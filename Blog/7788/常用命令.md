# 常用命令

1. git

    - init 
    - clone
    - add
    - status
    - diff
    - commit
    - reset
    - rm
    - branch
    - checkout
    - merge
    - log
    - tag
    - remote
    - fetch
    - pull
    - push


    git clone ...

    git add 

    git log --pretty=oneline

    git reset --merge   恢复到merge请求之前的

    git diff [文件名]   比较文件差异

    git branch -a   查看所有分支信息

    git branch -v   查看本地库中的所有分支

    git branch dev  创建dev分支

    git checkout dev    切换到dev分支

    git add [文件名] 添加待提交的文件

    git commit -m '日志信息' 提交
